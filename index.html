Introduction

What is Horus ?

Horus is just a micro php framework helps you to start your next web application .
Horus 5 doesn't like to has any complex code.
Horus 5 Comes with:
  • SQL Class Based on PDO
  • Events/Hooks
  • Simple universal class loader
  • Smart light uri router
  • Output control

Routes

A Route is a {HTTP-Method} + {URL pattern} + {Handler} .

  • HTTP Method: a thing like GET, POST, PUT, DELETE, ... etc
  • URL pattern: the pattern that match a url under associated http method and the handler
  • Handler: any valid php callback that will handle the pattern .
Examples: 


// load horus
require 'Horus.php';

// start it
$horus = new Horus;

// handle the pattern '/'
// under GET method
// using the anonymous function [or any php valid callback]
$horus->router->get('/', function(){
    print 'hello world';
});

// handle the pattern '/'
// under POST method
// using the anonymous function [or any php valid callback]
$horus->router->post('/', function(){
    print 'hello world';
});

// handle the pattern '/'
// under PUT method
// using the anonymous function [or any php valid callback]
$horus->router->put('/', function(){
    print 'hello world';
});

// handle the pattern '/'
// under DELETE method
// using the anonymous function [or any php valid callback]
$horus->router->delete('/', function(){
    print 'hello world';
});

// handle the pattern '/'
// under XYZ method [ any custom method ]
// using the anonymous function [or any php valid callback]
$horus->router->xyz('/', function(){
    print 'hello world';
});

// handle the pattern '/'
// under GET & POST method
// using the anonymous function [or any php valid callback]
$horus->router->get_post('/', function(){
    print 'hello world';
});

// handle the pattern '/'
// under GET, PUT & POST method
// using the anonymous function [or any php valid callback]
$horus->router->get_put_post('/', function(){
    print 'hello world';
});

// handle the pattern '/'
// under any method [get,post,put,..... all]
// using the anonymous function [or any php valid callback]
$horus->router->any('/', function(){
    print 'hello world';
});

# Route patterns may include regex parameters
$horus->router->any('/hello/([a-zA-Z0-9-_\.]+)', function($name){
    print "hello $name";
});

# instead of writing regex, there are some regex shortcuts in horus router
/*
'{num}'      =>  '([0-9\.,]+)',
'{alpha}'    =>  '([a-zA-Z]+)',
'{alnum}'    =>  '([a-zA-Z0-9\.]+)',
'{str}'      =>  '([a-zA-Z0-9-_\.]+)',
'{any}'      =>  '(.+)',
'{*}'        =>  '?|(.*?)'
*/

# You can add your own custom shortcuts
$horus->router->shortcut(':my:', '(.*?)');
# Or
    $this->shortcut(array(
        '{num}'      =>  '([0-9\.,]+)',
        '{alpha}'    =>  '([a-zA-Z]+)',
        '{alnum}'    =>  '([a-zA-Z0-9\.]+)',
        '{str}'      =>  '([a-zA-Z0-9-_\.]+)',
        '{any}'      =>  '(.+)',
        '{*}'        =>  '?|(.*?)'
    ));

// you can remove a shortcut
$horus->router->unshortcut(':my:');

# You can group some routes under one pattern
$horus->router->with('/admin', function() use($horus) {

	// route the 'admin/setting'
	$horus->router->any('/setting', function() {
		echo 'admin/setting';
	});

	// route the 'admin/posts'
	$horus->router->any('/posts', function() {
		echo 'admin/posts';
	});
	
});

# route multiple patterns at one
$horus->router->get( array( '/', 'index.php', 'home.py' ) , function(){
    echo "Home";
});

# check whether the current uri matches a pattern or not ?
if($horus->router->is('/admin/posts')) {
	// do something here
}

# if the url has any extension, it will be at end of the argument list
$horus->router->get('/i.php', function($ext){
    echo $ext; // php
});
$horus->router->get('/{str}/i.xml', function($string, $ext){
    echo "We are in /$string/i.xml , and ", "the extension is $ext";
    // ext: xml
});

// do you know that you can call another route from
// any route
// say we have this login router
$horus->router->post('/login', function(){
    var_dump($_POST);
});

// and we have this index router
// and we want to call login router without redirect !
$horus->router->any('/', function(){
    // get horus instance
    $horus = Horus::instance();
    
    // call login handler over HTTP POST
    // also you can pass array of params to it
    // and show it's response
    echo $horus->router->request
    (
        // first param is the request uri
        '/login', 
        // second param is request method (optional , default it 'GET') 
        'POST', 
        // third param is array of request params (optional)
        array('user'=>'mohammed', 'pass'=>'test')
    );
});

# get params
# e.g: /index.jsp?k1=v1&k2=v2
$horus->router->get('/index.jsp', function(){
    // get horus instance
    $horus = Horus::instance();
    
    // Horus::inputs() will fetch all application
    // inputs through any http methods
    // it will merge _POST + _GET + {any othe such as 'PUT, PATCH, ... etc'}
    
    // dump all inputs
    var_dump($horus->input());
    
    // get the k1
    echo $horus->input('k1');
    // get the k2
    echo $horus->input('k2');
    
    // by default if the requested key is not set 
    // it will return FALSE
    // you can choose what to return
    // returns 'not_set' if there is no k3
    echo $horus->input('k3', 'not_set');

});

// run it
$horus->run();

Helpers



// load horus
require 'Horus.php';

// start it
$horus = new Horus;

// make any class/interface/namespace in any directory
// loaded on demand
# $horus->autoload($dir, $ext = 'php');
# example:
// make the '\vendor\' directory be autoloaded with it's 
// files' extension 'php'
$horus->autoload('\vendor');
// or with extension '.class.php'
$horus->autoload('\vendor', 'class.php');

// hide errors 'error_reporting = 0' ?
$horus->debug(FALSE);

// show it again
$horus->debug(TRUE);

// listen listener(s) to an event
$horus->listen('tag.1', function(){
    echo 'event-1';
});

$horus->listen('tag.1', function($x){
    echo 'event-'.$x;
});

// dispatch / trigger event and pass args
$horus->trigger('tag.1', array(5));

// do you know that the last argument will contain 
// the value of last executed event !
// this is useful for filters

# get the direct full url to horus
echo URL;

# get the full route to horus [will contains /index.php/ or not 'auto detected']
echo ROUTE;

# horus version
echo HORUS_VERSION;

# When horus started ? [in float 'microtime']
echo HRUS_START;

# directory separator
echo DS;

# override 404 page ?
$horus->e404 = function(){
    echo 'error 404';
};

// run it
$horus->run();

$horus = new Horus;

// basic connection ( just like pdo )
$horus->sql->connect($dns, $user, $pass, $array_options);

$horus->sql->mysql($host, $db, $user, $pass);
$horus->sql->mariaDB($host, $db, $user, $pass);
$horus->sql->mssql($host, $db, $user, $pass);
$horus->sql->oracle($host, $db, $user, $pass);
$horus->sql->pgsql($host, $db, $user, $pass);
$horus->sql->sqlite($file, $prsistent = false);

// to get the current used driver
echo $horus->sql->driver();

// to get the current used database
echo $horus->sql->database();

// to do a query
// returns false if not-done or $horus->sql object if ok .
$horus->sql->query($sql_statment, $inputs_array);
// example:
$horus->sql->query('select * from tablename where mycolumn = ? and my2column  = ?', array('value1', 'value2'));

// it comes with some basic shortcut methods
// insert
$horus->sql->insert('tablename', array(
    // 1)- array of columns
    array('name', 'email', 'password'),
    
    // 2)- array(s) of inserts ! in just one query !!
    array('myname-1', 'mymail-1', 'mypass-1'),
    array('myname-2', 'mymail-2', 'mypass-2') // more inserts
));

// replace
$horus->sql->insert('tablename', array(
    // 1)- array of columns
    array('name', 'email', 'password'),
    
    // 2)- array(s) of inserts ! in just one query !!
    array('myname-1', 'mymail-1', 'mypass-1'),
    array('myname-2', 'mymail-2', 'mypass-2') // more inserts
), true); // 3rd param set to true in replace

// do you need simple insert/replace for single insert ?

    'name'  =>  'newname',
    'email' => 'newmail'

// delete

Routes 


// gzhandler
ob_start('ob_gzhandler');

// load horus
require 'Horus.php';

// start it
$horus = new Horus;

// connect to mysql db
$horus->sql->mysql('localhost', 'mydb', 'username', 'password');

// make url matching case-insensitive
$_SERVER['PATH_INFO'] = strtolower($_SERVER['PATH_INFO']);

// assign the table object
$horus->posts = $horus->sql_table->using('posts');

// handle the index [GET only] and view all posts
$horus->router->get('/', function() use($horus) {
    // fetch all posts
    $posts = $horus->posts;
    $posts->get_all();
    while( $row = $posts->fetch(PDO::FETCH_ASSOC) ):
        echo sprintf(' %d)- %s ', $row['ID'], $row['title']);
    endwhile;
});

// handle the post [GET only] and view one post
$horus->router->get('/pst/{num}', function($ID) use($horus) {
    // fetch all posts
    $post = $horus->posts->get_one('ID = ?', $ID);
    // must be found
    $post or exit('post id not found in db');
    // view it
    echo $post['title'], ' - ';
    echo $post['content'];
});

// minify the output
$horus->listen('horus.output.before', function($horus) {
    $horus->output = preg_replace('/\s+/', ' ', $horus->output);
});

// run it
$horus->run();
